private void RunScript( Mesh mesh, Box bb, IEnumerable<Point3d> pts, out object colored_mesh) { mesh.VertexColors.Clear(); Dictionary<Point3d, Color> pointColors = AssignPointColors(pts); foreach (var v in mesh.Vertices) { Color color = InterpolatedColor(v, pointColors); mesh.VertexColors.Add(color); } colored_mesh = mesh; } Dictionary<Point3d, Color> AssignPointColors(IEnumerable<Point3d> pts) { Dictionary<Point3d, Color> pointColors = new Dictionary<Point3d, Color>(); // Assign colors to points foreach (var point in pts) { // Example: Assign a random color to each point Color color = Color.FromArgb(new Random().Next(256), new Random().Next(10), new Random().Next(100)); pointColors.Add(point, color); } return pointColors; } Color InterpolatedColor(Point3d v, Dictionary<Point3d, Color> pointColors) { double maxDistance = 20.0; // radius of color // Calculate the color based on distance from each point double totalWeight = 0.0; double totalR = 0.0, totalG = 0.0, totalB = 0.0; foreach (var kvp in pointColors) { Point3d point = kvp.Key; Color pointColor = kvp.Value; double distance = v.DistanceTo(point); if (distance < maxDistance) { double weight = 1.0 - (distance / maxDistance); totalR += weight * pointColor.R; totalG += weight * pointColor.G; totalB += weight * pointColor.B; totalWeight += weight; } } // Normalize the color components int r = (int)(totalR / totalWeight); int g = (int)(totalG / totalWeight); int b = (int)(totalB / totalWeight); return Color.FromArgb(r, g, b); } }
